🔘 DETECCIÓN DE BOTÓN (con anti-rebote)

def boton_pulsado(btn):
    if hat.BTN(btn) == 0:
        time.sleep(0.05)        # anti-rebote
        if hat.BTN(btn) == 0:
            while hat.BTN(btn) == 0:
                pass           # espera suelta
            return True
    return False

# Esperar con timeout
def esperar_boton(btn, timeout=None):
    t0 = time.time()
    while True:
        if boton_pulsado(btn): return True
        if timeout and time.time()-t0 > timeout:
            return False
        time.sleep(0.01)

💡 PWM — INTENSIDAD POR DISTANCIA

# Tabla: distancia → intensidad
tabla = {0: 1.0, 1: 0.6, 2: 0.2}
#        ↑centro   ↑vecino  ↑borde

def pwm_distancia(pos, freq=100):
    for i, led in enumerate(LEDS):
        d = abs(i - pos)
        duty = tabla.get(d, 0.0)
        hat.PWMLED(led,
            frequency=freq,
            dutycycle=duty)

# Lista directa de intensidades
intensidades = [0.2, 0.6, 1.0, 0.6, 0.2]
for led, d in zip(LEDS, intensidades):
    hat.PWMLED(led, frequency=100, dutycycle=d)

🔢 CONTADOR + LEDs BINARIOS

# Pines del contador
LED_BIT1 = 24   # bit 1 (vale 2)
LED_BIT0 = 25   # bit 0 (vale 1)
contador = 0

def actualizar_contador(delta):
    global contador
    contador = max(0, min(contador+delta, 3))
    hat.LED(LED_BIT1, bool(contador & 0b10))
    hat.LED(LED_BIT0, bool(contador & 0b01))
    print(f"Contador: {contador}")

# Uso:
# actualizar_contador(+1)  → gana
# actualizar_contador(-1)  → pierde

# Tabla de referencia:
# 0 → bit1=OFF bit0=OFF
# 1 → bit1=OFF bit0=ON
# 2 → bit1=ON  bit0=OFF
# 3 → bit1=ON  bit0=ON
